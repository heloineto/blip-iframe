export default Layout(MDX_DATA.getApplication);

## Usage

The `getApplication` function is very useful to get information about the current or another bot.
It is very likely that the information you need is available in the application object.
(Be sure to check the `applicationJson` property).

## Example

```ts
import { blip } from 'blip-iframe';

const response = await blip.getApplication(); // Get info about the current bot

if (!response.success) {
  console.error(response.error); // An error happened during the request
  return;
}

console.log(response.data); // The application object
```

## Parameters

The `getApplication` function accepts the following parameters:

```ts
interface GetApplicationParams {
  /**
   * The shortName (aka identity) of the bot to get details for
   * Leave blank to get details for the current bot
   */
  shortName?: string;
}

getApplication(params: GetApplicationParams | undefined): Promise<GetApplicationResponse>;
```

## Return type

The `getApplication` function returns the following response:

```ts
export interface GetApplicationResponse {
  shortName: string;
  name: string;
  description: string;
  accessKey: string;
  imageUri?: string;
  template: string;
  tenantId: string;
  created: string;
  updated: string;
  hasPermission: boolean;
  emailOwner: string;
  applicationUserPermissionModel: {
    permissionClaim: number;
    permissionAction: number;
  }[];
  channels: unknown[];
  applicationJson: Record<string, unknown>;
  applicationDomainActivations?: unknown[];
}
```
