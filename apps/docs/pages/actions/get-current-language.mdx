import { Layout } from '@/layout';
import { MDX_DATA } from '@/mdx';

export default Layout(MDX_DATA.getCurrentLanguage);

## Usage

The `getCurrentLanguage` function returns the current language of the user account.
It is very like one of the following values: 'pt', 'en' or 'es'.

## Example

```ts
import { blip } from 'blip-iframe';

const response = await blip.getCurrentLanguage();

if (!response.success) {
  console.error(response.error); // An error happened during the request
  return;
}

console.log(response.data); // The language string, very likely 'pt', 'en' or 'es
```

## Parameters

The `getCurrentLanguage` function accepts no parameters.

## Return type

The `getApplication` function returns the following response:

```ts
export type GetCurrentLanguageResponse = string;
```
