import { Layout } from '@/layout';
import { MDX_DATA } from '@/mdx';

export default Layout(MDX_DATA.showModal);

## Usage

The `showModal` function is used to show a modal dialog on the Blip platform with a title, body, confirm button, and cancel button.
Modals are used to display important information or allow users to perform actions that require confirmation.

## Example

```ts
import { blip } from 'blip-iframe';

await blip.showModal({
  title: 'Delete Item',
  body: 'Are you sure you want to delete this item?',
  confirm: 'Delete',
  cancel: 'Cancel',
});
```

## Parameters

The `showModal` function accepts the following parameters:

```ts
interface ShowModalRequest {
  action: 'showModal';
  content: {
    title?: string;
    body: string;
    confirm: string;
    cancel: string;
  };
}
```

## Return type

The `showModal` function does not return any value.
